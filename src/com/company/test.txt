public int[] longestIncreasingSubsequence(int[]arr, int n){
        int curr_max = 1;
        int k = 0;
        int [] lengths = new int[n];
        int [] val = new int[n];
        for(int i=0; i < n; i++){
            lengths[i] = 1;
        }
        for(int i = 1; i < n; i++){
            for(int j = 0; j < i; j++){

              if(arr[i] > arr[j])
                  lengths[i] = Math.max(lengths[i], lengths[j]+1);
                    if(lengths[i] > lengths[i-1]){
                        val[k] = arr[i];
                    }

            }

            k++;
           curr_max = Math.max(lengths[i], curr_max);
        }
        return val;

    }
